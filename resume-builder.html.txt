    <script>
        let experienceCount = 0;
        let educationCount = 0;

        function addExperience() {
            const id = experienceCount++;
            const html = `
                <div class="entry" id="exp-${id}">
                    <button class="remove-btn" onclick="removeExperience(${id})">×</button>
                    <div class="form-group">
                        <label>Job Title</label>
                        <input type="text" data-exp-title="${id}" placeholder="Senior Developer" oninput="updatePreview()">
                    </div>
                    <div class="form-group">
                        <label>Company</label>
                        <input type="text" data-exp-company="${id}" placeholder="Tech Company" oninput="updatePreview()">
                    </div>
                    <div class="form-group">
                        <label>Duration</label>
                        <input type="text" data-exp-duration="${id}" placeholder="Jan 2020 - Present" oninput="updatePreview()">
                    </div>
                    <div class="form-group">
                        <label>Description</label>
                        <textarea data-exp-desc="${id}" placeholder="What did you accomplish?" oninput="updatePreview()"></textarea>
                    </div>
                </div>
            `;
            document.getElementById('experienceList').insertAdjacentHTML('beforeend', html);
        }

        function removeExperience(id) {
            const el = document.getElementById(`exp-${id}`);
            if (el) el.remove();
            updatePreview();
        }

        function addEducation() {
            const id = educationCount++;
            const html = `
                <div class="entry" id="edu-${id}">
                    <button class="remove-btn" onclick="removeEducation(${id})">×</button>
                    <div class="form-group">
                        <label>School/University</label>
                        <input type="text" data-edu-school="${id}" placeholder="University Name" oninput="updatePreview()">
                    </div>
                    <div class="form-group">
                        <label>Degree</label>
                        <input type="text" data-edu-degree="${id}" placeholder="Bachelor of Science in Computer Science" oninput="updatePreview()">
                    </div>
                    <div class="form-group">
                        <label>Graduation Year</label>
                        <input type="text" data-edu-year="${id}" placeholder="2022" oninput="updatePreview()">
                    </div>
                </div>
            `;
            document.getElementById('educationList').insertAdjacentHTML('beforeend', html);
        }

        function removeEducation(id) {
            const el = document.getElementById(`edu-${id}`);
            if (el) el.remove();
            updatePreview();
        }

        function updatePreview() {
            // Personal Info
            const name = document.getElementById('fullName').value || 'Your Name';
            const email = document.getElementById('email').value || 'email@example.com';
            const phone = document.getElementById('phone').value || '(555) 123-4567';
            const location = document.getElementById('location').value || 'Location';

            document.getElementById('previewName').textContent = name;
            document.getElementById('previewContact').textContent = `${email} | ${phone} | ${location}`;

            // Summary
            const summary = document.getElementById('summary').value;
            document.getElementById('previewSummary').innerHTML = summary ? `
                <div class="preview-section-title">Professional Summary</div>
                <div class="preview-description">${summary}</div>
            ` : '';

            // Experience
            const expInputs = document.querySelectorAll('[data-exp-title]');
            let expHtml = '';
            if (expInputs.length > 0) {
                expHtml = '<div class="preview-section-title">Work Experience</div>';
                for (let i = 0; i < experienceCount; i++) {
                    const title = document.querySelector(`[data-exp-title="${i}"]`);
                    if (title && title.value) {
                        const company = document.querySelector(`[data-exp-company="${i}"]`).value;
                        const duration = document.querySelector(`[data-exp-duration="${i}"]`).value;
                        const desc = document.querySelector(`[data-exp-desc="${i}"]`).value;
                        expHtml += `
                            <div class="preview-entry">
                                <div class="preview-entry-title">${title.value}</div>
                                <div class="preview-entry-subtitle">${company}</div>
                                <div class="preview-entry-date">${duration}</div>
                                ${desc ? `<div class="preview-description">${desc}</div>` : ''}
                            </div>
                        `;
                    }
                }
            }
            document.getElementById('previewExperience').innerHTML = expHtml;

            // Education
            const eduInputs = document.querySelectorAll('[data-edu-school]');
            let eduHtml = '';
            if (eduInputs.length > 0) {
                eduHtml = '<div class="preview-section-title">Education</div>';
                for (let i = 0; i < educationCount; i++) {
                    const school = document.querySelector(`[data-edu-school="${i}"]`);
                    if (school && school.value) {
                        const degree = document.querySelector(`[data-edu-degree="${i}"]`).value;
                        const year = document.querySelector(`[data-edu-year="${i}"]`).value;
                        eduHtml += `
                            <div class="preview-entry">
                                <div class="preview-entry-title">${school.value}</div>
                                <div class="preview-entry-subtitle">${degree}</div>
                                <div class="preview-entry-date">${year}</div>
                            </div>
                        `;
                    }
                }
            }
            document.getElementById('previewEducation').innerHTML = eduHtml;

            // Skills
            const skills = document.getElementById('skills').value;
            document.getElementById('previewSkills').innerHTML = skills ? `
                <div class="preview-section-title">Skills</div>
                <div class="preview-description">${skills.split(',').map(s => s.trim()).join(' • ')}</div>
            ` : '';
        }

        function downloadPDF() {
            // Show paywall modal
            alert('📥 Download your resume as PDF!\n\n✨ Free users can preview only.\n💎 Upgrade to download and export.');
            // Later this will trigger Stripe checkout
        }

        // Initialize
        updatePreview();
    </script>
